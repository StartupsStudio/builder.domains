// Example: Domain rental marketplace
class DomainRentalMarketplace {
  async rentDomain(criteria: RentalCriteria) {
    // Find available domain matching criteria
    const domain = await this.findAvailableDomain({
      keyword: criteria.keyword,
      tld: criteria.tld,
      seo_score: { min: criteria.min_seo_score },
      available_hours: criteria.duration_hours
    });

    // Rent domain for specified duration
    const rental = await this.createRental({
      domain,
      renter: criteria.user_id,
      duration: criteria.duration_hours,
      price: this.calculateRentalPrice(domain, criteria.duration_hours)
    });

    // Configure domain for renter
    await domain.configure({
      owner: rental.renter,
      target: rental.deployment_url,
      expires: Date.now() + (criteria.duration_hours * 3600000)
    });

    return rental;
  }

  // After rental expires, domain returns to pool
  async handleRentalExpiry(rental: Rental) {
    await rental.domain.reset();
    await this.returnToPool(rental.domain);
  }
}
